#### 陣列 (array)
內容盡量同一類型
```js
let array=["Appale","Bear","Cat","Duck","Egg"];
```

###### 陣列特性
取值
```js
console.log(array[3]); //Duck
```
令值
```js
array[0]="Monkey"; 
console.log(array[0]); //Monkey
```
新增值
```js
array.push(true);      
console.log(array); //["Monkey","Bear","Cat","Duck","Egg",true]
```
長度取得
```js
console.log(array.length); //6
```
轉換
```js
let array=[1,2,3,4,5];
let map=arr5.map( n=>n*2 ); //[2,4,6,8,10]
```
過濾
```js
let array=[1,2,3,4,5];
let filter=array.filter( n=>n>3 );//[4,5]
```
#### 物件 (object)
key必為字串
```js
let object={
    "name":"Lin",   //string
    "age":30,       //int
    "other":[1,2,3],//array
    "msg":{         //object
        "math":80,
        "eng":99
    }
};
```

###### 物件特性
取值
```js
console.log(object["name"]); //Lin
```
令值
```js
object["age"]=18; //30=>18
console.log(object["age"]); //18
```
新增值
```js
object["sign"]=true;
```
列出物件全部key 
```js
console.log(Object.keys(object)); //["name","age","other","msg"]
```
列出物件全部value
```js
console.log(Object.keys(object)); 
//["Lin",30,[1,2,3],{"math":80,"eng":99},ture]
```

#### 時間資料 (Date)
```js
console.log(new Date()); //2025-09-15T12:37:08.911Z
```
###### 時間資料特性
獲取特定時間資料
```js
console.log(new Date().getFullYear()); //2025
```
